./presto.jar --server localhost:8080 --catalog cassandra --schema default --output-format CSV_HEADER --execute "SELECT * FROM cassandra.ks_imdb.imdb;" --output-format CSV_HEADER > out_alldata.csv
./presto.jar --server localhost:8080 --catalog cassandra --schema default --output-format CSV_HEADER --execute "SELECT original_language,count(original_language) AS count FROM cassandra.ks_imdb.imdb GROUP BY original_language ORDER BY count DESC;" --output-format CSV_HEADER > rank_original_language.csv
./presto.jar --server localhost:8080 --catalog cassandra --schema default --output-format CSV_HEADER --execute "SELECT original_language, ROUND(AVG(vote_average), 2) AS avg_vote FROM cassandra.ks_imdb.imdb GROUP BY original_language ORDER BY avg_vote DESC;" --output-format CSV_HEADER > rank_lang_vote.csv
